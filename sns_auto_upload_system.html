youtube_uploader.py 
import os
from google.oauth2.credentials import Credentials
from google_auth_oauthlib.flow import InstalledAppFlow
from googleapiclient.discovery import build
from googleapiclient.http import MediaFileUpload
from dotenv import load_dotenv

load_dotenv()

SCOPES = ["https://www.googleapis.com/auth/youtube.upload"]

def get_youtube_service():
    # 최초 실행 시 브라우저 인증 → token.json 저장
    if os.path.exists("token.json"):
        creds = Credentials.from_authorized_user_file("token.json", SCOPES)
    else:
        flow = InstalledAppFlow.from_client_secrets_file(
            "client_secrets.json", SCOPES
        )
        creds = flow.run_local_server(port=0)
        with open("token.json", "w") as f:
            f.write(creds.to_json())
    
    return build("youtube", "v3", credentials=creds)


def upload_to_youtube(video_path, title, description, tags=[], is_short=False):
    youtube = get_youtube_service()
    
    # Shorts는 #Shorts 태그 + 60초 이하
    if is_short:
        title = title + " #Shorts"
        description = "#Shorts\n\n" + description
    
    request = youtube.videos().insert(
        part="snippet,status",
        body={
            "snippet": {
                "title": title,
                "description": description,
                "tags": tags,
                "categoryId": "22",
                "defaultLanguage": "ko",
            },
            "status": {
                "privacyStatus": "public",
                "selfDeclaredMadeForKids": False,
            }
        },
        media_body=MediaFileUpload(
            video_path,
            mimetype="video/mp4",
            resumable=True   # 대용량 영상도 안전하게
        )
    )
    
    response = request.execute()
    video_id = response["id"]
    print(f"✅ YouTube 업로드 완료: https://youtu.be/{video_id}")
    return video_id
facebook_uploader.py 
import requests
import os
from dotenv import load_dotenv

load_dotenv()

PAGE_ID = os.getenv("META_PAGE_ID")
ACCESS_TOKEN = os.getenv("META_ACCESS_TOKEN")
GRAPH_URL = "https://graph.facebook.com/v18.0"


def upload_to_facebook(video_path, description, title):
    """Facebook 페이지에 영상 업로드"""
    url = f"{GRAPH_URL}/{PAGE_ID}/videos"
    
    with open(video_path, "rb") as video_file:
        response = requests.post(
            url,
            data={
                "description": description,
                "title": title,
                "access_token": ACCESS_TOKEN,
            },
            files={"source": video_file}
        )
    
    result = response.json()
    
    if "id" in result:
        print(f"✅ Facebook 업로드 완료: {result['id']}")
        return result["id"]
    else:
        print(f"❌ Facebook 업로드 실패: {result}")
        raise Exception(result.get("error", {}).get("message"))


# ─── 대용량 영상용 (Resumable Upload) ─────────────────
def upload_large_video_facebook(video_path, description):
    # Phase 1: 업로드 세션 시작
    file_size = os.path.getsize(video_path)
    
    init_response = requests.post(
        f"{GRAPH_URL}/{PAGE_ID}/videos",
        data={
            "upload_phase": "start",
            "file_size": file_size,
            "access_token": ACCESS_TOKEN,
        }
    ).json()
    
    upload_session_id = init_response["upload_session_id"]
    
    # Phase 2: 파일 전송 (청크 단위)
    chunk_size = 10 * 1024 * 1024  # 10MB
    offset = 0
    
    with open(video_path, "rb") as f:
        while chunk := f.read(chunk_size):
            requests.post(
                f"{GRAPH_URL}/{PAGE_ID}/videos",
                data={
                    "upload_phase": "transfer",
                    "upload_session_id": upload_session_id,
                    "start_offset": offset,
                    "access_token": ACCESS_TOKEN,
                },
                files={"video_file_chunk": chunk}
            )
            offset += len(chunk)
    
    # Phase 3: 완료
    finish = requests.post(
        f"{GRAPH_URL}/{PAGE_ID}/videos",
        data={
            "upload_phase": "finish",
            "upload_session_id": upload_session_id,
            "description": description,
            "access_token": ACCESS_TOKEN,
        }
    ).json()
    
    print(f"✅ Facebook 대용량 업로드 완료: {finish}")
    return finish
instagram_uploader.py (Graph API) 
import requests
import os
import time
from dotenv import load_dotenv

load_dotenv()

IG_USER_ID = os.getenv("IG_USER_ID")
ACCESS_TOKEN = os.getenv("META_ACCESS_TOKEN")
GRAPH_URL = "https://graph.facebook.com/v18.0"

# ⚠️ 중요: 영상은 공개 URL이 필요함!
# → S3, GCS, 또는 임시 공개 URL 사용

def upload_reel_to_instagram(video_url, caption):
    """Instagram Reels 업로드 (2단계)"""
    
    # Step 1: 미디어 컨테이너 생성
    container_response = requests.post(
        f"{GRAPH_URL}/{IG_USER_ID}/media",
        data={
            "media_type": "REELS",
            "video_url": video_url,      # 공개 접근 가능한 URL!
            "caption": caption,
            "share_to_feed": "true",
            "access_token": ACCESS_TOKEN,
        }
    ).json()
    
    container_id = container_response.get("id")
    if not container_id:
        raise Exception(f"컨테이너 생성 실패: {container_response}")
    
    print(f"📦 컨테이너 생성: {container_id} → 처리 대기 중...")
    
    # Step 2: 처리 완료까지 폴링 (최대 5분)
    for _ in range(30):
        time.sleep(10)
        status_response = requests.get(
            f"{GRAPH_URL}/{container_id}",
            params={
                "fields": "status_code",
                "access_token": ACCESS_TOKEN,
            }
        ).json()
        
        status = status_response.get("status_code")
        
        if status == "FINISHED":
            break
        elif status == "ERROR":
            raise Exception("영상 처리 실패")
        print(f"  ⏳ 처리 중... ({status})")
    
    # Step 3: 게시
    publish_response = requests.post(
        f"{GRAPH_URL}/{IG_USER_ID}/media_publish",
        data={
            "creation_id": container_id,
            "access_token": ACCESS_TOKEN,
        }
    ).json()
    
    media_id = publish_response.get("id")
    print(f"✅ Instagram Reels 업로드 완료: {media_id}")
    return media_id

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.options import Options
from webdriver_manager.chrome import ChromeDriverManager
import time, os

# ⚠️ 주의: Selenium은 TikTok 정책상 언제든 막힐 수 있음
# 공식 API 승인 후에는 아래 공식 방법 사용 권장

def upload_to_tiktok_selenium(video_path, caption):
    """Selenium으로 TikTok 업로드 (반자동)"""
    options = Options()
    options.add_argument("--user-data-dir=./tiktok_profile")  # 로그인 유지
    options.add_argument("--disable-blink-features=AutomationControlled")
    options.add_experimental_option("excludeSwitches", ["enable-automation"])
    
    driver = webdriver.Chrome(
        ChromeDriverManager().install(),
        options=options
    )
    wait = WebDriverWait(driver, 20)
    
    try:
        driver.get("https://www.tiktok.com/upload")
        time.sleep(3)
        
        # 파일 업로드 input 찾기
        file_input = wait.until(
            EC.presence_of_element_located((By.XPATH, "//input[@type='file']"))
        )
        file_input.send_keys(os.path.abspath(video_path))
        
        print("📤 파일 업로드 중...")
        time.sleep(10)  # 업로드 대기
        
        # 캡션 입력
        caption_input = wait.until(
            EC.presence_of_element_located((
                By.CSS_SELECTOR, "div[contenteditable='true']"
            ))
        )
        caption_input.clear()
        caption_input.send_keys(caption)
        
        # 게시 버튼 클릭
        post_btn = wait.until(
            EC.element_to_be_clickable((
                By.XPATH, "//button[contains(text(), 'Post')]"
            ))
        )
        post_btn.click()
        
        time.sleep(5)
        print("✅ TikTok 업로드 완료!")
        
    finally:
        driver.quit()


# ─── 공식 API 방식 (Content Posting API 승인 후) ────────
def upload_to_tiktok_api(video_path, caption):
    """공식 TikTok Content Posting API"""
    ACCESS_TOKEN = os.getenv("TIKTOK_ACCESS_TOKEN")
    
    # 1단계: 업로드 초기화
    init_response = requests.post(
        "https://open.tiktokapis.com/v2/post/publish/video/init/",
        headers={
            "Authorization": f"Bearer {ACCESS_TOKEN}",
            "Content-Type": "application/json",
        },
        json={
            "post_info": {
                "title": caption,
                "privacy_level": "PUBLIC_TO_EVERYONE",
            },
            "source_info": {
                "source": "FILE_UPLOAD",
                "video_size": os.path.getsize(video_path),
                "chunk_size": 10 * 1024 * 1024,
                "total_chunk_count": 1,
            }
        }
    ).json()
    
    print(f"TikTok 업로드 세션: {init_response}")

import os
import schedule
import time
from datetime import datetime

from youtube_uploader import upload_to_youtube
from facebook_uploader import upload_to_facebook
from instagram_uploader import upload_reel_to_instagram
from tiktok_uploader import upload_to_tiktok_selenium


def multi_upload(video_path, title, description, tags=[]):
    """영상 1개 → 모든 SNS 자동 업로드"""
    results = {}
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M")
    
    print(f"\n🚀 멀티 업로드 시작: {timestamp}")
    print(f"   영상: {video_path}")
    print(f"   제목: {title}")
    print("─" * 50)
    
    # 1. YouTube Shorts 업로드
    try:
        yt_id = upload_to_youtube(
            video_path, title, description,
            tags=tags, is_short=True
        )
        results["youtube"] = {"success": True, "id": yt_id}
    except Exception as e:
        results["youtube"] = {"success": False, "error": str(e)}
        print(f"❌ YouTube 실패: {e}")
    
    # 2. Facebook 업로드
    try:
        fb_id = upload_to_facebook(video_path, description, title)
        results["facebook"] = {"success": True, "id": fb_id}
    except Exception as e:
        results["facebook"] = {"success": False, "error": str(e)}
        print(f"❌ Facebook 실패: {e}")
    
    # 3. Instagram Reels (공개 URL 필요)
    try:
        ig_caption = f"{title}\n\n{description}\n\n" + " ".join([f"#{t}" for t in tags])
        # video_path → CDN URL로 변환 필요 (S3/GCS 등)
        video_url = upload_to_cdn(video_path)  # 별도 구현
        ig_id = upload_reel_to_instagram(video_url, ig_caption)
        results["instagram"] = {"success": True, "id": ig_id}
    except Exception as e:
        results["instagram"] = {"success": False, "error": str(e)}
        print(f"❌ Instagram 실패: {e}")
    
    # 4. TikTok (Selenium)
    tiktok_caption = f"{title} " + " ".join([f"#{t}" for t in tags])
    try:
        upload_to_tiktok_selenium(video_path, tiktok_caption)
        results["tiktok"] = {"success": True}
    except Exception as e:
        results["tiktok"] = {"success": False, "error": str(e)}
        print(f"❌ TikTok 실패: {e}")
    
    # 결과 요약
    print("\n📊 업로드 결과 요약")
    print("─" * 50)
    for platform, result in results.items():
        status = "✅" if result["success"] else "❌"
        print(f"{status} {platform.upper():15} {result}")
    
    return results


# ─── 스케줄러 (매일 오전 9시 자동 업로드) ──────────────
def scheduled_upload():
    # 업로드할 영상 큐 (DB나 파일로 관리 가능)
    queue = [
        {
            "path": "videos/today_short.mp4",
            "title": "오늘의 팁 5가지",
            "desc": "유용한 일상 팁을 알려드립니다",
            "tags": ["팁", "일상", "쇼츠"],
        }
    ]
    for item in queue:
        multi_upload(item["path"], item["title"], item["desc"], item["tags"])


# 매일 09:00에 자동 실행
schedule.every().day.at("09:00").do(scheduled_upload)
